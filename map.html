var currentProv = null;

function selectProv(name){

  var p = PROV_MAP[name];

  if(!p) return;

  if(currentProv){

    var oldRow = document.getElementById('row-'+currentProv);

    if(oldRow) oldRow.classList.remove('sel');

  }

  currentProv = name;

  var row = document.getElementById('row-'+name);

  if(row){ row.classList.add('sel'); row.scrollIntoView({behavior:'smooth',block:'nearest'}); }

  document.getElementById('sel-name').textContent = name;

  document.getElementById('sel-region').textContent = p.region;

  var btn = document.getElementById('jump-btn');

  if(p.link){

    btn.href = p.link;

    btn.classList.remove('disabled');

    btn.onclick = null;

  } else {

    btn.href = '#';

    btn.classList.add('disabled');

    btn.onclick = function(e){

      e.preventDefault();

      alert('请在代码顶部 PROVINCES 数组中\n找到 id:"'+name+'" 的条目，填入 link 字段');

    };

  }

  chart.dispatchAction({type:'select',seriesIndex:0,name:name});

}
